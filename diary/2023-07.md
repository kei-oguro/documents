using System.Collections.Generic;
using System.Linq;
using System.Threading.Tasks;

class C
{
    private static async ValueTask WhenAll(ICollection<(ValueTask task, int wait)> list)
    {
        Console.WriteLine($"{list.Count()} tasks will start.");

        foreach (var (item, i) in list.Select((x, i) => (x, i)))
        {
            Console.WriteLine($"  start {i} for {item.wait}. Thread id: {Thread.CurrentThread.ManagedThreadId}");
            await item.task;
        }
    }

    public static void Main()
    {
        static async ValueTask Delay(int i)
        {

            .Select(x => (6 - x) * 1000)
            .Select(x => (Delay(x), x))

}

<details><summary>WhenAll()の実装。</summary>

```cs
/* This prints...
5 tasks will start.
100
200
300
400
500
finished.
*/
using System;
using System.Collections.Generic;
using System.Linq;
using System.Threading.Tasks;
class C {
    private static async ValueTask WhenAll(IReadOnlyList<ValueTask> tasks) {
        Console.WriteLine($"{tasks.Count()} tasks will start.");
        for (var i = 0; i < tasks.Count(); ++i) {
            await tasks[i].ConfigureAwait(false);  // すぐにTaskが開始されて、ここでJoinしない。
        }
        // 全ての開始したTaskがここで合流する。
    }
    public static void Main() {
        static async ValueTask delay(int i) {
            await Task.Delay(i*100);
            Console.WriteLine(i*100);
        };
        WhenAll(Enumerable.Range(1, 5)
            .Select(x => delay(6 - x))
            .ToList()
        ).AsTask().Wait();
        Console.WriteLine("finished.");
    }
}
```
